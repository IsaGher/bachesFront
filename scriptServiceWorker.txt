Script para intalar un serviceWorker en el navegador, se a√±ade al index.html

<script src="NetworkListener.js"></script>
  <script>
    const registrar = async ()=>{
    if('serviceWorker' in navigator){
      try{
        const registro = await navigator.serviceWorker.register("./NetworkListener.js");
        if(registro.installing){
          console.log("Instalando");
        }else if(registro.active){
          console.log("Activo");
        }
      }catch(error){
        console.error(`No se pudo registrar ${error}`);
      }
    }
  }
  registrar();
</script>
